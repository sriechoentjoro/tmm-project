    function selectAddress($model, $field_prop, $field_kab, $field_kec = null, $field_kel = null) {

        $html = '';
        $link_propinsi = FULL_BASE_URL . $this->Html->url(
                        array(
                            'controller' => 'daerahs',
                            'action' => 'optionPropinsi'
                        )
        );
        $link_kabupaten = FULL_BASE_URL . $this->Html->url(
                        array(
                            'controller' => 'daerahs',
                            'action' => 'optionKabupaten'
                        )
        );
        $link_kecamatan = FULL_BASE_URL . $this->Html->url(
                        array(
                            'controller' => 'daerahs',
                            'action' => 'optionKecamatan'
                        )
        );
        $link_kelurahan = FULL_BASE_URL . $this->Html->url(
                        array(
                            'controller' => 'daerahs',
                            'action' => 'optionKelurahan'
                        )
        );
        $id_form = Inflector::humanize($model . ucfirst($this->params['action']) . 'Form');
        $id_propinsi = Inflector::classify($model . Inflector::humanize($field_prop));
        $id_kabupaten = Inflector::classify($model . Inflector::humanize($field_kab));
        $id_kecamatan = Inflector::classify($model . Inflector::humanize($field_kec));
        $id_kelurahan = Inflector::classify($model . Inflector::humanize($field_kel));

        return

                $this->Packer->packJS(
                        <<<EOD
$(document).ready(function () {
    $("#{$id_form} #{$id_propinsi}").change(function () {
        var id_propinsi = $(this).val();

        $.ajax({
            type: "POST",
            url: "$link_kabupaten" + "/id_propinsi:" + id_propinsi,
            cache: false,
            success: function (html) {
                $("#{$id_form} #{$id_kabupaten}").html(html);
            }
        });
    });
    $("#{$id_form} #{$id_kabupaten}").change(function () {
        var id_propinsi = $("#{$id_form} #{$id_propinsi}").val();
        var id_kabupaten = $(this).val();

        $.ajax({
            type: "POST",
            url: "$link_kecamatan" + "/id_propinsi:" + id_propinsi + "/id_kabupaten:" + id_kabupaten,
            cache: false,
            success: function (html) {
                $("#{$id_form} #{$id_kecamatan}").html(html);
            }
        });
    });
    $("#{$id_form} #{$id_kecamatan}").change(function () {
        var id_propinsi = $("#{$id_form} #{$id_propinsi}").val();
        var id_kabupaten = $("#{$id_form} #{$id_kabupaten}").val();
        var id_kecamatan = $(this).val();

        $.ajax({
            type: "POST",
            url: "$link_kelurahan" + "/id_propinsi:" + id_propinsi + "/id_kabupaten:" + id_kabupaten + "/id_kecamatan:" + id_kecamatan,
            cache: false,
            success: function (html) {
                $("#{$id_form} #{$id_kelurahan}").html(html);
            }
        });
    });
});
EOD
        );
    }
